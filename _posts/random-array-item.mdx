## Usando o `Match.random()`

```typescript
const randomArrayItem = <T>(array: T[]): T => {
  return array[Math.floor(Math.random() * array.length)];
};
``` 
```typescript
let arr: number[] = [1, 2, 3, 4, 5];

if (arr.length === 0) {
  console.log(undefined);
} else {
  const ind: number = Math.floor(Math.random() * arr.length);
  const result: number = arr[ind];
  console.log(`Random Element = ${result}`);
}
```

## Usando o `splice()`

```typescript
let arr: number[] = [1, 2, 3, 4, 5, 6, 8, 7];

if (arr.length === 0) {
  console.log(undefined);
} else {
  const ind = Math.floor(Math.random() * arr.length);
  const result = arr.splice(ind, 1)[0];
  console.log(`Random Element = ${result}`);
}

```
